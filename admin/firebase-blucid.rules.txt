// firestore.rules
// Deploy: firebase deploy --only firestore:rules
//
// Rules:
//   - Admin (authenticated) can read/write EVERYTHING
//   - Public (unauthenticated) can only READ services, products, websiteContent
//   - Public can CREATE inquiries (contact form submissions)
//   - Job orders and inquiries are admin-only to read

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Helper: is the user a logged-in admin?
    function isAdmin() {
      return request.auth != null;
    }

    // Services — public can read, only admin can write
    match /services/{docId} {
      allow read: if true;
      allow write: if isAdmin();
    }

    // Products — public can read, only admin can write
    match /products/{docId} {
      allow read: if true;
      allow write: if isAdmin();
    }

    // Website content (hero, about, contact info) — public read, admin write
    match /websiteContent/{docId} {
      allow read: if true;
      allow write: if isAdmin();
    }

    // Job Orders — admin only (private business data)
    match /jobOrders/{docId} {
      allow read, write: if isAdmin();
    }

    // Customer Inquiries — public can create (contact form), admin can read/delete
    match /inquiries/{docId} {
      allow create: if true;            // anyone can submit a contact form
      allow read, update, delete: if isAdmin();
    }
  }
}


// storage.rules
// Deploy: firebase deploy --only storage

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {

    // Service images — public read, admin write
    match /services/{allPaths=**} {
      allow read: if true;
      allow write: if request.auth != null
                   && request.resource.size < 10 * 1024 * 1024;
    }

    // Product images — public read, admin write
    match /products/{allPaths=**} {
      allow read: if true;
      allow write: if request.auth != null
                   && request.resource.size < 10 * 1024 * 1024;
    }

    // Website content images (hero bg, about image) — public read, admin write
    match /websiteContent/{allPaths=**} {
      allow read: if true;
      allow write: if request.auth != null
                   && request.resource.size < 10 * 1024 * 1024;
    }
  }
}
